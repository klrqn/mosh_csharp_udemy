using System;

namespace properties
{
    public class Person
    {
        private DateTime _birthdate;
        
        public void SetBirthdate(DateTime birthdate)
        {
            _birthdate = birthdate;
        }
        
        public DateTime GetBirthDate()
        {
            return _birthdate;
        }
    }
}
